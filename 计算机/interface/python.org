** 统计文本 字数

  #+begin_src python
    from pathlib import Path
    
    filename="python.org"
    
    path=Path(filename)
    
    try:
        contents=path.read_text(encoding='utf-8')
    except FileNotFoundError:
        pass
    else:
        words=contents.split()
        print(len(words))
#+end_src
